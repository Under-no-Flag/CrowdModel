
```mermaid
graph TD
    %% -----------------------------------------------------------
    %% 1. 输入与场景定义层 (Input & Scenario Definition)
    %% -----------------------------------------------------------
    subgraph Layer1 ["输入与场景定义层 (Inputs & Configuration)"]
        direction TB
        Geometry[/"1. 场景几何与离散网格:<br/>外滩观景平台边界,<br/>1-10号通道几何坐标 (x,y)"/]

        GuidanceInfo[/"2. 几何引导信息:<br/>各通道内切向向量场 tau(x),<br/>度量张量系数 alpha, beta"/]

        MgtRules[/"3. 管理管控管控措施接口:<br/>指示函数 chi_c, 符号 s_c<br/>e.g., 4号单向上, 7号单往下"/]

        RoutePlan[/"4. 游客路线与分流设置:<br/>阶段定义 S={1:进场, 2:游览, 3:离场...}<br/>各阶段目标区 Gs, 阶段切换率 kappa_s<br/>固定分流概率 p_s->r (e.g., 去8, 9, 10号通道)"/]

        PhysParams[/"5. 物理参数:<br/>最大速度 vmax, 最大密度 rho_max,<br/>时间步长 dt, 空间步长 dx"/]
    end

    %% -----------------------------------------------------------
    %% 2. 核心数学模型引擎 (Core Mathematical Engine)
    %% -----------------------------------------------------------
    subgraph Layer2 ["核心数学模型引擎 (Mathematical Engine)"]
        direction TB

        SpeedFunc["速度大小函数:<br/>f(rho) = vmax * (1 - rho/rho_max)"]

        AnisotropicMet["各向异性度量构建:<br/>M(x) = alpha*tau*tauT + beta*n*nT<br/>(通道内 alpha >> beta)"]

        DirConstraint["方向约束集合构建 (单向通行):<br/>U(x) = { tau } (单向内)<br/>U(x) = { |u|=1 } (普通区域)"]

        %% 核心 HJB 方程
        HJB_Eq{{"约束各向异性 Bellman (HJB) 方程求解:<br/>为每个(阶段s, 路线r)求解 phi_s,r(x)"}}
        HJB_Formula["phi = min_{u in U} [ phi(x+du) + du/(f(rho)*sqrt(uTuMu)) ]"]
        HJB_Eq -.- HJB_Formula

        OptDir["最优方向恢复:<br/>u*_s,r = argmin (Bellman右端项)"]

        GroupVel["群体速度场:<br/>v_s,r(x) = f(rho) * u*_s,r(x)"]

        ContinuityEq{{"连续性方程 (守恒格式)求解:<br/>更新 rho_s,r(x,t)"}}
        PDE_Formula["drho/dt + div(rho * v) = Q_in - Q_out"]
        ContinuityEq -.- PDE_Formula

        FixedSplit[["固定概率分流与阶段切换处理:<br/>若到達Gs: Delta rho = p_s->r * kappa * chi * rho"]]
    end

    %% -----------------------------------------------------------
    %% 3. 计算流程与循环 (Computational Workflow Loop)
    %% -----------------------------------------------------------
    subgraph Layer3 ["计算流程与循环 (Workflow Loop per Time Step)"]
        direction LR
        Initialize[初始化 rho_s,r] --> TimeLoop{时间步循环 t -> t+dt}

        TimeLoop --> StepA1(A. 计算当前总密度 rho = sum_s sum_r rho_s,r)
        StepA1 --> StepA2(A. 更新全局速度大小 f<rho>)

        StepA2 --> StepB1[/"B. 针对每个群体(s,r)进行计算Loop"/]

        subgraph GroupUpdateLoop ["群体s,r状态更新"]
            StepB1 --> CalcPhi("B1. 解 Bellman HJB 方程:<br/>计算 phi_s,r")
            CalcPhi --> CalcVel("B2. 恢复最优方向,<br/>计算 v_s,r")
            CalcVel --> UpdDensity("B3. 解连续性方程:<br/>平流更新 rho_s,r")
        end

        UpdDensity --> StepC1("C. 执行分流:<br/>基于固定概率进行阶段/路线转移")
        StepC1 --> SinkMass[处理 Sink 项 (目标区出口)]
        SinkMass --> CheckEnd{达到总时间?}
        CheckEnd -- No --> TimeLoop
        CheckEnd -- Yes --> Finish[模拟完成]
    end

    %% -----------------------------------------------------------
    %% 4. 输出与评估 (Outputs & Evaluation)
    %% -----------------------------------------------------------
    subgraph Layer4 ["输出与评估层 (Outputs)"]
        Finish --> PlotDens[密度分布热力图<br/>(总密度 rho, 分群体 rho_s,r)]
        Finish --> PlotVel[速度场/流线图]
        Finish --> EvalMetrics[管控措施评估:<br/>通道流量, 总疏散时间,<br/>安全风险区域 (rho>阈值)]
    end

    %% -----------------------------------------------------------
    %% 依赖与控制流连接 (Dependencies)
    %% -----------------------------------------------------------
    Geometry --> AnisotropicMet
    GuidanceInfo --> AnisotropicMet
    MgtRules --> DirConstraint
    MgtRules --> GuidanceInfo
    RoutePlan --> HJB_Eq
    RoutePlan --> FixedSplit
    PhysParams --> SpeedFunc

    %% 数学引擎内部连接
    AnisotropicMet --> HJB_Eq
    DirConstraint --> HJB_Eq
    SpeedFunc --> HJB_Eq
    HJB_Eq --> OptDir
    OptDir --> GroupVel
    GroupVel --> ContinuityEq

    %% 循环与引擎连接
    StepA2 ==依赖==> SpeedFunc
    CalcPhi ==执行==> HJB_Eq
    CalcVel ==执行==> OptDir
    UpdDensity ==执行==> ContinuityEq
    StepC1 ==依赖==> RoutePlan
```
